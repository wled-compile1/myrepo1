#pragma once

#include "wled.h"
#include <SPI.h>

class P4SpiOutputUsermod : public Usermod {
private:
  // ===== CONFIG =====
  static constexpr uint32_t SPI_CLOCK = 20000000; // 20 MHz
  static constexpr uint8_t  PIN_CS    = 10;       // Adjust to your board
  static constexpr uint8_t  PIN_SCK   = 12;
  static constexpr uint8_t  PIN_MOSI  = 11;

  SPIClass *spi = nullptr;
  uint8_t frameId = 0;
  bool initialized = false;

public:
  void setup() override {
    if (initialized) return;

    pinMode(PIN_CS, OUTPUT);
    digitalWrite(PIN_CS, HIGH);

    spi = new SPIClass(FSPI);
    spi->begin(PIN_SCK, -1, PIN_MOSI, PIN_CS);

    initialized = true;
    DEBUG_PRINTLN(F("P4 SPI Output Usermod initialized"));
  }

  void loop() override {
    // Nothing needed here
  }

  /**
   * This is the KEY hook.
   * Called after WLED finishes rendering all segments.
   */
  void onUpdate() override {
    if (!initialized || strip.getLengthTotal() == 0) return;

    uint16_t ledCount = strip.getLengthTotal();
    uint8_t *rgb = strip.getPixels(); // GRB internally, WLED converts

    spi->beginTransaction(SPISettings(SPI_CLOCK, MSBFIRST, SPI_MODE0));
    digitalWrite(PIN_CS, LOW);

    // ---- Frame header ----
    spi->transfer(0xAA);
    spi->transfer(0x55);
    spi->transfer(frameId++);
    spi->transfer(ledCount & 0xFF);
    spi->transfer(ledCount >> 8);

    // ---- Pixel data ----
    // WLED buffer is already RGB order here
    spi->writeBytes(rgb, ledCount * 3);

    digitalWrite(PIN_CS, HIGH);
    spi->endTransaction();
  }

  uint16_t getId() override {
    return USERMOD_ID_UNSPECIFIED;
  }
};
