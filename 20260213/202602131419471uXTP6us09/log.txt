class HexWavesUsermod : public Usermod {
  private:
    // Unikalne ID efektu, aby nie gryzło się z fabrycznymi
    uint8_t effectId = 0;
    const char* _name = "Hex Waves 2D";

  public:
    void setup() {
      // Rejestracja efektu w systemie WLED
      // Nazwa wyświetli się na liście efektów
      effectId = strip.addEffect(255, mode_hex_waves, _name);
    }

    void loop() {
      // Tu nic nie musimy robić, logika jest w funkcji efektu
    }

    // To jest serce Twojego efektu
    static uint16_t mode_hex_waves() {
      // Pobieramy dane o aktualnym segmencie
      WS2812FX::Segment& seg = strip.getSegment(strip.getCurrSegId());
      
      uint16_t width  = seg.virtualWidth();
      uint16_t height = seg.virtualHeight();
      
      // t obliczane na podstawie suwaka Speed
      uint32_t t = millis() * (seg.speed / 64.0);

      for (uint16_t y = 0; y < height; y++) {
        for (uint16_t x = 0; x < width; x++) {
          
          // Obliczamy dystans od środka Twojego heksagonu (4,4)
          float dx = (float)x - 4.0f;
          float dy = (float)y - 4.0f;
          float dist = sqrtf(dx*dx + dy*dy);
          
          // Suwak Intensity (ix) kontroluje zagęszczenie fal
          float scale = seg.intensity / 128.0f;
          float wave = sinf(dist * scale - (t * 0.05f));
          
          // Mapowanie fali na jasność
          uint8_t bright = (wave + 1.0f) * 127.5f;

          // Ustawiamy kolor (wykorzystuje Color 1 i Color 2 z palety)
          uint32_t color = color_blend(BLACK, seg.colors[0], bright);
          seg.setPixelColorXY(x, y, color);
        }
      }
      return FRAMETIME;
    }

    uint16_t getId() { return USERMOD_ID_RESERVED; }
};
