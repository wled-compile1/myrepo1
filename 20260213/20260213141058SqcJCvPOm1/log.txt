    uint16_t mode_hex_waves() {
      uint16_t width  = SEGMENT.virtualWidth();
      uint16_t height = SEGMENT.virtualHeight();
      uint32_t t = millis() * (SEGMENT.speed / 50.0);

      for (uint16_t y = 0; y < height; y++) {
        for (uint16_t x = 0; x < width; x++) {
          float dx = x - 4.0;
          float dy = y - 4.0;
          float dist = sqrt(dx*dx + dy*dy);
          float wave = sin(dist * 0.8 - (t * 0.01));
          uint8_t bright = (wave + 1.0) * 127.5;
          SEGMENT.setPixelColorXY(x, y, color_blend(BLACK, SEGCOLOR(0), bright));
        }
      }
      return FRAMETIME;
    }
